{
parameters:
@signal[arrival](type="long");
@statistic[hopCount](title="hop count"; source="arrival"; record=vector,stats; interpolationmode=none);
@display("i=block/routing");
gates:
inout gate[];
}

{
parameters:
@figure[description](type=text; pos=5,20; font=,,bold;
